#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cout<<"enter the size of array : ";
    cin>>n;
    int arr[n];
    //input
    for(int i=0;i<=n-1;i++){
        cin>>arr[i];
    }
    //Duplicate
    bool flag=false;
    for(int i=0; i<=n-1; i++){
        for(int j=i+1; j<=n-1; j++){
            if(arr[i]==arr[j]){
                flag=true;
                cout<<"yes it is duplicate" <<" "<<arr[i];
                break;
            }
        }
    }
    if (flag==false){
        cout<<"no duplicate";
    }
    return 0;
}
